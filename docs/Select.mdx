# Select

Renders a select input that will accept a list of options and emit values.

## Basic usage

<Playground>
  {() => {
    const [value, setValue] = React.useState('Pizza');

    return (
      <div>
        <p>
          <Select
            value={value}
            onChange={setValue}
            options={['Pizza', 'Spaghetti']}
          />
        </p>

        <p>
          <Select
            value={value}
            onChange={setValue}
            options={[
              { label: 'Pizza', value: 'Pizza' },
              { label: 'Spaghetti', value: 'Spaghetti' }
            ]}
          />
        </p>

        <pre><code>VALUE: {JSON.stringify(value)}</code></pre>
      </div>
    );

}}

</Playground>
