# FileInput

Renders a `FileInput` FileInput that will values.

## Basic usage

<Playground>
  {() => {
    const [file, setFile] = React.useState(null);
    const [files, setFiles] = React.useState([]);

    return (
      <div>
        <div>
          <FileInput onChange={setFile} />{" "}
          <pre><code>VALUE: {JSON.stringify(file ? file.name : null)}</code></pre>
        </div>

        <div>
          <FileInput multiple onChange={setFiles} />{" "}
          <pre><code>VALUE: {JSON.stringify(Array.from(files).map(f => f.name))}</code></pre>
        </div>
      </div>
    );

}}

</Playground>
